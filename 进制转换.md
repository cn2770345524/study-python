```java
# -*-coding:gb2312-*-

# 二进制
a = 0b101
# 八进制
b = 0o17
# 十进制
c = 10
# 十六进制
d = 0x1a

# 将10进制转为16进制的字符串
print(hex(c))
# 将10进制转为8进制的字符串
print(oct(c))
# 将10进制转为2进制的字符串
print(bin(c))

# 将二进制字符串转换为int类型的变量
int('0b101', 2)
# 将八进制字符串转换为int类型的变量
int('0o12', 8)
# 将十进制字符串转换为int类型的变量
int('15')
# 将16进制字符串转换为int类型的变量
int('0x1a', 16)
```
